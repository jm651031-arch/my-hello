// ══════════════════════════════════════════ // 工具：顯示提示 (優化捲動體驗) // ══════════════════════════════════════════ function showAlert(containerId, type, icon, msg) { const container = document.getElementById(containerId); container.innerHTML = `
${icon}${msg}
`; // 改為 'center'，確保畫面會精準地移動並讓訊息置中，避免被上方導航列遮住 container.scrollIntoView({ behavior: 'smooth', block: 'center' }); } // ══════════════════════════════════════════ // 報名送出 (優化 fetch 連線，繞過 CORS 限制) // ══════════════════════════════════════════ async function submitForm() { const name = document.getElementById('f-name').value.trim(); const year = document.getElementById('f-year').value; const month = document.getElementById('f-month').value; const day = document.getElementById('f-day').value; const phone = document.getElementById('f-phone').value.trim().replace(/-/g, ''); const idNum = document.getElementById('f-id').value.trim().toUpperCase(); const payment = document.querySelector('input[name="payment"]:checked'); // 驗證 if (!name || name.replace(/\s/g,'').length < 2) return showAlert('signup-alert', 'error', '⚠️', '請填寫真實姓名（至少2個字）'); if (!year || !month || !day) return showAlert('signup-alert', 'error', '⚠️', '請選擇完整出生年月日'); const cleanPhone = phone.replace(/[^0-9]/g,''); if (!cleanPhone.startsWith('09') || cleanPhone.length !== 10) return showAlert('signup-alert', 'error', '⚠️', '電話格式錯誤，請輸入 09 開頭共 10 碼數字'); if (!validateID(idNum)) return showAlert('signup-alert', 'error', '⚠️', '身分證字號格式錯誤
正確格式：第1碼英文、第2碼1或2、後8碼數字（例：A123456789）'); if (!payment) return showAlert('signup-alert', 'error', '⚠️', '請選擇付款方式'); const btn = document.getElementById('submit-btn'); btn.disabled = true; btn.innerHTML = '送出中…'; const birthday = `${year}/${String(month).padStart(2,'0')}/${String(day).padStart(2,'0')}`; const payload = { action: 'signup', name, birthday, phone, idNum, payment: payment.value }; try { const res = await fetch(GAS_URL, { method: 'POST', // 加入 headers 將 Content-Type 設為 text/plain，繞過 GAS 的 CORS 預檢限制 headers: { 'Content-Type': 'text/plain;charset=utf-8', }, redirect: 'follow', // 允許跟隨 GAS 的重新導向 body: JSON.stringify(payload) }); const data = await res.json(); if (data.status === 'duplicate') { showAlert('signup-alert', 'error', '🚫', `此身分證字號已報名過，無需重複報名！
如有疑問請聯絡主辦單位。`); } else if (data.status === 'success') { showAlert('signup-alert', 'success', '🎉', `${name} 報名成功！
我們將盡速確認您的資料，期待6月20日與您相見 🌿`); // 清空表單 document.getElementById('f-name').value = ''; document.getElementById('f-year').value = ''; document.getElementById('f-month').value = ''; document.getElementById('f-day').value = ''; document.getElementById('f-phone').value = ''; document.getElementById('f-id').value = ''; document.querySelectorAll('input[name="payment"]').forEach(r => r.checked = false); } else { showAlert('signup-alert', 'error', '❌', '送出失敗，請稍後再試或聯絡主辦單位。'); } } catch(e) { console.error(e); showAlert('signup-alert', 'error', '❌', '連線失敗，請確認網路後再試。'); } btn.disabled = false; btn.innerHTML = '✅ 確認送出報名'; } // ══════════════════════════════════════════ // 查詢 (同樣補上 headers 繞過限制) // ══════════════════════════════════════════ async function queryMember() { const val = document.getElementById('q-input').value.trim(); if (!val) return; document.getElementById('query-result').innerHTML = `
查詢中…
`; try { const res = await fetch(GAS_URL, { method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8', }, redirect: 'follow', body: JSON.stringify({ action: 'query', keyword: val }) }); const data = await res.json(); if (data.status === 'found') { const m = data.member; const payClass = m.payment === 'Line Pay' ? 'linepay' : m.payment === '轉帳' ? 'transfer' : 'cash'; document.getElementById('query-result').innerHTML = `

姓名${m.name}

電話${m.phone}

`; } else { document.getElementById('query-result').innerHTML = `
