function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("報名名單"); if (!sheet) sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var data = JSON.parse(e.postData.contents); var action = data.action; // 1. 處理報名 if (action === 'signup') { var dataRange = sheet.getDataRange().getValues(); for (var i = 1; i < dataRange.length; i++) { if (dataRange[i][4] === data.idNum) { return ContentService.createTextOutput(JSON.stringify({status: 'duplicate'})).setMimeType(ContentService.MimeType.JSON); } } // 🪄 將餐點 (data.meal) 放在第 8 欄 (H 欄)，避開原有的付款公式欄位 sheet.appendRow([new Date(), data.name, data.birthday, "'" + data.phone, data.idNum, data.payment, '', data.meal]); return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON); } // 2. 處理查詢 if (action === 'query') { var dataRange = sheet.getDataRange().getValues(); for (var i = 1; i < dataRange.length; i++) { if (String(dataRange[i][1]) === data.keyword || String(dataRange[i][3]) === data.keyword) { var m = { timestamp: dataRange[i][0], name: dataRange[i][1], birthday: dataRange[i][2], phone: String(dataRange[i][3]), payment: dataRange[i][5], isPaid: dataRange[i][6] || '', meal: dataRange[i][7] || '未填寫' // 讀取餐點 }; return ContentService.createTextOutput(JSON.stringify({status: 'found', member: m})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({status: 'not_found'})).setMimeType(ContentService.MimeType.JSON); } // 3. 處理讀取名單 if (action === 'list') { var dataRange = sheet.getDataRange().getValues(); var members = []; for (var i = 1; i < dataRange.length; i++) { if (dataRange[i][1]) { members.push({ timestamp: dataRange[i][0], name: dataRange[i][1], birthday: dataRange[i][2], phone: String(dataRange[i][3]), payment: dataRange[i][5], isPaid: dataRange[i][6] || '', meal: dataRange[i][7] || '未填寫' // 讀取餐點 }); } } return ContentService.createTextOutput(JSON.stringify({status: 'success', members: members})).setMimeType(ContentService.MimeType.JSON); } // 4. 處理更新付款狀態 if (action === 'updatePayment') { var dataRange = sheet.getDataRange().getValues(); for (var i = 1; i < dataRange.length; i++) { if (String(dataRange[i][3]) === String(data.phone)) { sheet.getRange(i + 1, 7).setValue(data.status); return ContentService.createTextOutput(JSON.stringify({status: 'success'})).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({status: 'error'})).setMimeType(ContentService.MimeType.JSON); } }
